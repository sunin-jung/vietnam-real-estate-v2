🔧 이미지 문제 해결 도구

이 페이지는 베트남 부동산 플랫폼의 이미지 로딩 문제를 해결하는 도구입니다.

수동 해결 방법

개발자 도구 콘솔에서 다음 코드를 실행하세요:

// 이미지 문제 해결 const fixImages = () => { const stored = localStorage.getItem('vietnam-properties'); if (stored) { const properties = JSON.parse(stored); const defaultImages = { 'Apartment': 'https://images.unsplash.com/photo-1545324418-cc1a3fa10c00?w=500', 'House': 'https://images.unsplash.com/photo-1560518883-ce09059eeffa?w=500', 'Villa': 'https://images.unsplash.com/photo-1613977257363-707ba9348227?w=500', 'Office': 'https://images.unsplash.com/photo-1486406146926-c627a92ad1ab?w=500', 'Shop': 'https://images.unsplash.com/photo-1570129477492-45c003edd2be?w=500', 'Land': 'https://images.unsplash.com/photo-1500382017468-9049fed747ef?w=500', 'Other': 'https://images.unsplash.com/photo-1545324418-cc1a3fa10c00?w=500' }; const fixedProperties = properties.map(property => { if (!property.images || property.images.length === 0) { const defaultImage = defaultImages[property.property_type] || defaultImages['Other']; return { ...property, images: [defaultImage] }; } const validImages = property.images.filter(img => { if (!img) return false; if (img.startsWith('blob:')) return false; if (!img.startsWith('http')) return false; return true; }); if (validImages.length === 0) { const defaultImage = defaultImages[property.property_type] || defaultImages['Other']; return { ...property, images: [defaultImage] }; } return { ...property, images: validImages }; }); localStorage.setItem('vietnam-properties', JSON.stringify(fixedProperties)); console.log('이미지 문제가 해결되었습니다!'); location.reload(); } }; fixImages();
